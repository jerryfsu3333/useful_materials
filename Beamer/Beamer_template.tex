\lyxframe{Sufficient dimension reduction (SDR) and central subspace (CS)}

\structure{Variables:}
\begin{itemize}
	\small
	\item $Y\in\mbbR$: \structure{response}
	\item $\mbX\in\mbbR^{p}$: \structure{predictor}
\end{itemize}

%{\small \structure{Key idea:} By projecting $\mbX$ on some low-dimensional subspace, we lose no information on the conditional distribution of $Y|\mbX$.}

\structure{Definition:} 
\begin{itemize}
	\small
	\item A subspace $\spn(\mbB)$, the span of the basis matrix $\mbB \in \mbbR^{p\times d}, d < p$, is called a \structure{central subspace (CS)} if it is the smallest subspace such that
	\begin{equation*}
	Y \indep \mbX | \mbB^T \mbX \quad \Longleftrightarrow \quad Y|\mbX \sim Y|\mbB^T\mbX.
	\end{equation*}
	And it is denoted by $\mathcal{S}_{Y | \mbX}$.
	\item By projecting $\mbX$ onto $\mathrm{span}(\mbB)$, the dimension is reduced from $p$ to $d$, which is called \structure{Sufficent Dimension Reduction (SDR)}.
\end{itemize}

\lyxframe{Subspace distance}

\begin{itemize}
	{\small
		\item \structure{Subspace distance:} For the true central subspace $\mbB$ and the estimator $\hatmbB$,
		\begin{align*}
		D(\mbB, \hatmbB) = \|\mbP_{\mbB} - \mbP_{\mbB}\mbP_{\hatmbB}\mbP_{\mbB}\|_F/\sqrt{d},
		\end{align*}
		where $d$ is the dimension of true subspace $\mbB$.
	}
\end{itemize}
